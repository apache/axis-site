18  package org.apache.savan.publication;
19  
20  import java.util.ArrayList;
21  import java.util.Hashtable;
22  
23  import org.apache.savan.SavanException;
24  
25  /***
26   * This will encapsulate error information of a specific publication.
27   * Probably will contain details of each subscriber to which the message could not
28   * be delivered successfully. 
29   */
30  public class PublicationReport {
31  
32  	/***
33  	 * The susbscribers to which this msg could not be sent. Probably their ID and the
34  	 * Exception that occured.
35  	 */
36  	private Hashtable errors = null;
37  	
38  	/***
39  	 * Ids of the subscribers to which this msg could be sent successfully.
40  	 */
41  	private ArrayList notifiedSubscribers;
42  	
43  	public PublicationReport () {
44  		errors = new Hashtable ();
45  		notifiedSubscribers = new ArrayList ();
46  	}
47  	
48  	public void addErrorReportEntry (String id, SavanException reason) {
49  		errors.put(id,reason);
50  	}
51  	
52  	public void addNotifiedSubscriber (String subscriberID) {
53  		notifiedSubscribers.add(subscriberID);
54  	}
55  
56  	public Hashtable getErrors() {
57  		return errors;
58  	}
59  
60  	public ArrayList getNotifiedSubscribers() {
61  		return notifiedSubscribers;
62  	}
63  	
64  	
65  }
