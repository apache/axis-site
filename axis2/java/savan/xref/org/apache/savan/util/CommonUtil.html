44  	public static Calendar addDurationToCalendar (Calendar calendar,Duration duration) {
45  		calendar.add(Calendar.YEAR,duration.getYears());
46  		calendar.add(Calendar.MONTH,duration.getMonths());
47  		calendar.add(Calendar.DATE,duration.getDays());
48  		calendar.add(Calendar.HOUR,duration.getHours());
49  		calendar.add(Calendar.MINUTE,duration.getMinutes());
50  		calendar.add(Calendar.SECOND,(int) duration.getSeconds());
51  		
52  		return calendar;
53  	}

55  	/***
56  	 * Will be used by test cases to load XML files from test-resources as Envelopes
57  	 * SOAP 1.1 is assumed
58  	 * 
59  	 * @param path
60  	 * @param name
61  	 * @return
62  	 */
63  	public static SOAPEnvelope getTestEnvelopeFromFile (String path, String name) throws IOException {
64          try {
65          	String fullName = path + File.separator + name;
66              FileReader reader = new FileReader(fullName);
67              XMLStreamReader streamReader = XMLInputFactory.newInstance().createXMLStreamReader(
68                      reader);
69              StAXSOAPModelBuilder builder = OMXMLBuilderFactory.createStAXSOAPModelBuilder(
70                      OMAbstractFactory.getSOAP11Factory(), streamReader);
71              return builder.getSOAPEnvelope();
72          } catch (XMLStreamException e) {
73              throw new RuntimeException(e);
74          }
75  	}
76  	
77  	public static boolean isDuration (String timeStr) {
78          return timeStr.startsWith("p") || timeStr.startsWith("P") || timeStr.startsWith("-p") || timeStr.startsWith("-P");
79  
80      }

82  	public static SubscriberStore getSubscriberStore (AxisService axisService) {
83  		Parameter parameter = axisService.getParameter(SavanConstants.SUBSCRIBER_STORE);
84  		if (parameter==null)
85  			return null;
86  		
87  		return (SubscriberStore) parameter.getValue();
88  	}
