18  package org.apache.savan;
19  
20  import org.apache.axiom.soap.SOAPEnvelope;
21  import org.apache.axis2.AxisFault;
22  import org.apache.axis2.context.ConfigurationContext;
23  import org.apache.axis2.context.MessageContext;
24  import org.apache.axis2.description.Parameter;
25  import org.apache.savan.configuration.Protocol;
26  import org.apache.savan.storage.SubscriberStore;
27  
28  /***
29   * This encaptulates a Axis2 Message Context.
30   * Provide some easy methods to access Savan specific properties easily.
31   */
32  public class SavanMessageContext {
33  
34  	MessageContext messageContext = null;
35  	
36  	public SavanMessageContext (MessageContext messageContext) {
37  		this.messageContext = messageContext;
38  	}
39  	
40  	public void setMessageType (int type) {
41  		messageContext.setProperty(SavanConstants.MESSAGE_TYPE, new Integer (type));
42  	}
43  	
44  	public int getMessageType () {
45  		Integer typeInt = (Integer) messageContext.getProperty(SavanConstants.MESSAGE_TYPE);
46  		if (typeInt==null) {
47  			typeInt = new Integer (SavanConstants.MessageTypes.UNKNOWN);
48  			messageContext.setProperty(SavanConstants.MESSAGE_TYPE,typeInt);
49  		}
50  		
51  		return typeInt.intValue();
52  	}
53  	
54  	public ConfigurationContext getConfigurationContext () {
55  		return messageContext.getConfigurationContext();
56  	}
57  	
58  	public Object getProperty (String key) {
59  		return messageContext.getProperty(key);
60  	}
61  	
62  	public void setProperty (String key, Object val) {
63  		messageContext.setProperty(key,val);
64  	}
65  	
66  	public SOAPEnvelope getEnvelope () {
67  		return messageContext.getEnvelope();
68  	}
69  	
70  	public MessageContext getMessageContext () {
71  		return messageContext;
72  	}
73  	
74  	public SubscriberStore getSubscriberStore () {
75  		Parameter parameter = messageContext.getParameter(SavanConstants.SUBSCRIBER_STORE);
76  		SubscriberStore subscriberStore = null;
77  		if (parameter!=null) {
78  			parameter = messageContext.getParameter(SavanConstants.SUBSCRIBER_STORE);
79  			subscriberStore = (SubscriberStore) parameter.getValue();
80  		}
81  		
82  		return subscriberStore;
83  	}
84  	
85  	public void setSubscriberStore (SubscriberStore store) throws SavanException  {
86  		Parameter parameter = new Parameter ();
87  		parameter.setName(SavanConstants.SUBSCRIBER_STORE);
88  		parameter.setValue(store);
89  		
90  		try {
91  			messageContext.getAxisService().addParameter(parameter);
92  		} catch (AxisFault e) {
93  			String message = "Could not add the AbstractSubscriber Store parameter";
94  			throw new SavanException (message,e);
95  		}
96  	}
97  	
98  	public void setProtocol (Protocol protocol) {
99  		messageContext.setProperty(SavanConstants.PROTOCOL, protocol);
100 	}
101 	
102 	public Protocol getProtocol () {
103 		return (Protocol) messageContext.getProperty(SavanConstants.PROTOCOL);
104 	}
105 	
106 }
